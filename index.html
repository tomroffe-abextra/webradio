    <script>

        const player = document.getElementById('radio-player');

        const playButton = document.getElementById('play-button');

        const playIcon = document.getElementById('play-icon');

        const liveIndicator = document.getElementById('live-indicator');


        function updateButton() {

            if (player.paused) {

                playIcon.innerHTML = '<path d="M8 5v14l11-7z"/>'; // Play icon

                liveIndicator.style.display = 'none';

            } else {

                playIcon.innerHTML = '<path d="M6 6h4v12H6zm8 0h4v12h-4z"/>'; // Pause icon

                liveIndicator.style.display = 'inline-block';

            }

        }


        playButton.addEventListener('click', function() {

            if (player.paused) {

                player.src = player.src; // Recharge le flux pour éviter le blocage

                player.play().catch(error => console.log("Lecture bloquée", error));

            } else {

                player.pause();

            }

            updateButton();

        });


        // Corrige l'affichage initial du bouton (autoplay actif = pause affiché)

        window.onload = function() {

            updateButton();

        };
